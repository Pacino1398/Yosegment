YOSEGMENT RK3588（Ubuntu aarch64）板端部署/运行指令（rknnlite）

# 必须在仓库根目录执行（否则可能 No module named 'app'）
cd /opt/Yosegment

1) 系统依赖
sudo apt update
sudo apt install -y python3-pip python3-opencv

2) Python 依赖
python3 -m pip install -U pip
python3 -m pip install -U numpy pyyaml matplotlib
# 无桌面环境（推荐）
python3 -m pip install -U opencv-python-headless
# 有桌面环境需要 --view 时改用：
# python3 -m pip uninstall -y opencv-python-headless
# python3 -m pip install -U opencv-python

3) 安装 RKNN 板端运行时（rknnlite）
python3 -m pip install -U rknn-toolkit-lite2

4) 验证 rknnlite 可用
python3 -c "from rknnlite.api import RKNNLite; print('RKNNLite import OK')"

5) 准备文件（相对仓库根目录）
ls -lh data/my.yaml
ls -lh weights/0414_qy++.rknn

6) 最小闭环（单图）
python3 -m tools.test_rknn_infer \
  --rknn weights/0414_qy++.rknn \
  --image test_input/000000.jpg \
  --imgsz 640

# 如果你的 .rknn 是 NHWC uint8 输入：
# python3 -m tools.test_rknn_infer \
#   --rknn weights/0414_qy++.rknn \
#   --image test_input/000000.jpg \
#   --imgsz 640 \
#   --input-format nhwc_u8

7) 实时运行（RKNN + 路径规划）
# 摄像头（无桌面环境不要加 --view）
python3 -m app.planning.realtime_pathplan \
  --source 0 \
  --weights ./weights/0414_qy++.rknn \
  --backend rknn \
  --rknn-input-format nchw_fp32 \
  --data ./data/my.yaml

# 有桌面环境显示窗口：
 python3 -m app.planning.realtime_pathplan \
   --source 0 \
   --weights ./weights/0414_qy++.rknn \
   --backend rknn \
   --rknn-input-format nchw_fp32 \
   --data ./data/my.yaml \
   --view
